CentOS7配置静态IP地址

2020-12-20 Linux 313 字

实验环境为Centos7.5

  • 打开配置文件
[root@localhost ~]# cd ..
[root@localhost /]# ls
bin  boot  dev  etc  home  lib  lib64  media  mnt  opt  proc  root  run  sbin  srv  sys  tmp  usr  var
[root@localhost /]# vi etc/sysconfig/network-scripts/ifcfg-ens33
  • 编辑配置文件
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=statics #dhcp为动态获取IP地址，statics为静态获取IP地址。
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=ens33
UUID=4b20da7e-0986-46f4-8e98-ab3b16fc4f0e
DEVICE=ens33
ONBOOT=yes #是否开机启用该网卡no为否，yes为是。

#在后方添加一下
IPADDR=192.168.10.10 #IP地址设置

NETMASK=255.255.255.0 #子网掩码设置

GETWAY=192.168.10.1 #网关设置
#保存退出：ESC Shift+: 输入:wq #保存并退出
  • 重启网卡
[root@localhost /]# systemctl restart network
测试
